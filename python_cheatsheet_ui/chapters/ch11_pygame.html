Empty Game Window

import sys
import pygame

pygame.init()
screen = pygame.display.set_mode((1200, 800))
pygame.display.set_caption("Alien Invasion")
clock = pygame.time.Clock()

while True:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            sys.exit()
    pygame.display.flip()
    clock.tick(60)
Opens a 1200x800 game window titled "Alien Invasion"

Working with Images

ship = pygame.image.load('images/ship.bmp')
ship_rect = ship.get_rect()
screen_rect = screen.get_rect()
ship_rect.midbottom = screen_rect.midbottom
screen.blit(ship, ship_rect) # draw to screen
Ship image loaded and positioned at bottom center

Keyboard Input

for event in pygame.event.get():
    if event.type == pygame.KEYDOWN:
        if event.key == pygame.K_RIGHT:
            ship_rect.x += 1
        elif event.key == pygame.K_LEFT:
            ship_rect.x -= 1
        elif event.key == pygame.K_SPACE:
            ship.fire_bullet()
        elif event.key == pygame.K_q:
            sys.exit()
← → moves ship, SPACE fires, Q quits
