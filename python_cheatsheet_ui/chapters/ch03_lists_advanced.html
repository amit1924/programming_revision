Simple Statistics

ages = [93, 99, 66, 17, 85]
print(f"Youngest: {min(ages)}, Oldest: {max(ages)}, Sum: {sum(ages)}")
Youngest: 17, Oldest: 99, Sum: 360

Slicing

finishers = ['sam', 'bob', 'ada', 'bea', 'zoe']
print(finishers[:2]) # first two
['sam', 'bob']
print(finishers[1:4]) # middle three
['bob', 'ada', 'bea']
print(finishers[-3:]) # last three
['ada', 'bea', 'zoe']

Copying a List

bikes = ['trek', 'redline', 'giant']
copy_of_bikes = bikes[:]
print(copy_of_bikes)
['trek', 'redline', 'giant']
Always use [:] to copy — otherwise changes affect both!

List Comprehensions

# Loop approach
squares = []
for x in range(1, 6):
    squares.append(x**2)
print(squares)
[1, 4, 9, 16, 25]
# Comprehension (shorter!)
squares = [x**2 for x in range(1, 6)]
print(squares)
[1, 4, 9, 16, 25]

Upper Case Comprehension

names = ['kai', 'abe', 'ada', 'gus', 'zoe']
upper_names = [name.upper() for name in names]
print(upper_names)
['KAI', 'ABE', 'ADA', 'GUS', 'ZOE']
