Passing Arguments

def greet_user(username):
    print(f"Hello, {username}!")
greet_user('jesse')
Hello, jesse!

Positional & Keyword Args

def describe_pet(animal, name):
    print(f"I have a {animal} named {name}.")
describe_pet('hamster', 'harry')
I have a hamster named harry.
describe_pet(animal='dog', name='willie')
I have a dog named willie.

Default Values

def describe_pet(name, animal='dog'):
    print(f"I have a {animal}.")
describe_pet('willie')
I have a dog.

Return Values

def get_full_name(first, last):
    return f"{first} {last}".title()
musician = get_full_name('jimi', 'hendrix')
print(musician)
Jimi Hendrix

Return a Dictionary

def build_person(first, last, age=None):
    person = {'first': first, 'last': last}
    if age: person['age'] = age
    return person
print(build_person('jimi', 'hendrix', 27))
{'first': 'jimi', 'last': 'hendrix', 'age': 27}

Passing a List

def greet_users(names):
    for name in names:
        print(f"Hello, {name}!")
greet_users(['hannah', 'ty', 'margot'])
Hello, hannah!
          Hello, ty!
          Hello, margot!

Modifying a List

def print_models(unprinted, printed):
    while unprinted:
        current = unprinted.pop()
        print(f"Printing {current}")
        printed.append(current)
unprinted = ['phone case', 'pendant', 'ring']
printed = []
print_models(unprinted, printed)
print(f"Printed: {printed}")
Printing ring
          Printing pendant
          Printing phone case
          Printed: ['ring', 'pendant', 'phone case']

Arbitrary Args (*args)

def make_pizza(size, *toppings):
    print(f"{size} pizza with:")
    for t in toppings:
        print(f"- {t}")
make_pizza('large', 'bacon', 'pineapple')
large pizza with:
          - bacon
          - pineapple

Arbitrary Keyword Args (**kwargs)

def build_profile(first, last, **info):
    info['first'] = first
    info['last'] = last
    return info
user = build_profile('albert', 'einstein', location='princeton')
print(user)
{'location': 'princeton', 'first': 'albert', 'last': 'einstein'}
