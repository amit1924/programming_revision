Testing a Class

# account.py
class Account:
    def __init__(self, balance=0):
        self.balance = balance
    def deposit(self, amount):
        self.balance += amount

# test_account.py
def test_initial_balance():
    account = Account()
    assert account.balance == 0

def test_deposit():
    account = Account()
    account.deposit(100)
    assert account.balance == 100
