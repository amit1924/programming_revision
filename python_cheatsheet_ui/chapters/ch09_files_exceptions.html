Path Objects

from pathlib import Path
path = Path("text_files/alice.txt")
print(path.name) # filename
print(path.suffix) # extension
print(path.exists()) # exists?

Reading Files

path = Path('siddhartha.txt')
contents = path.read_text()
print(contents[:50]) # first 50 chars
(file contents appear here)
lines = contents.splitlines()

Writing to Files

path = Path('programming.txt')
msg = "I love programming!"
path.write_text(msg)
print(path.read_text())
I love programming!
# Append by reading first, then rewriting
contents = path.read_text()
contents += "\nI love Python."
path.write_text(contents)
print(path.read_text())
I love programming!
          I love Python.

Try-Except

try:
    print(5 / 0)
except ZeroDivisionError:
    print("Can't divide by zero!")
Can't divide by zero!
path = Path('missing.txt')
try:
    contents = path.read_text()
except FileNotFoundError:
    print(f"Can't find {path}")
Can't find missing.txt

Try-Except-Else

x, y = 10, 2
try:
    result = x / y
except ZeroDivisionError:
    print("Can't divide!")
else:
    print(result) # only if no error
5.0

Failing Silently

f_names = ['alice.txt', 'siddhartha.txt']
for f in f_names:
    path = Path(f)
    try:
        contents = path.read_text()
    except FileNotFoundError:
        pass # silently skip
    else:
        print(f"{f} found!")

JSON — Storing Data

import json
from pathlib import Path

numbers = [2, 3, 5, 7, 11]
path = Path('numbers.json')
path.write_text(json.dumps(numbers))

contents = path.read_text()
loaded = json.loads(contents)
print(loaded)
[2, 3, 5, 7, 11]

Best Practices

  • Use specific exception types (not bare except:)
  • Use except Exception as a catch-all if needed
  • Only put error-prone code in try
  • Put dependent code in else
  • pass in except = fail silently
