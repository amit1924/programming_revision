While Loops

current = 1
while current <= 5:
    print(current, end=' ')
    current += 1
1 2 3 4 5

Using a Flag

active = True
count = 0
while active:
    count += 1
    print(count, end=' ')
    if count >= 3: active = False
1 2 3

Break & Continue

count = 0
while count < 5:
    count += 1
    if count == 3: break
    print(count, end=' ')
1 2
count = 0
while count < 5:
    count += 1
    if count == 3: continue
    print(count, end=' ')
1 2 4 5

Remove All Instances

pets = ['dog', 'cat', 'fish', 'cat', 'rabbit', 'cat']
while 'cat' in pets:
    pets.remove('cat')
print(pets)
['dog', 'fish', 'rabbit']
