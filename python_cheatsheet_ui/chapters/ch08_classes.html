Classes

Creating & Using a Class

class Car:
    def __init__(self, make, model, year):
        self.make = make
        self.model = model
        self.year = year
        self.fuel_level = 0
    def fill_tank(self):
        self.fuel_level = 100
        print("Fuel tank full!")
my_car = Car('audi', 'a4', 2024)
print(my_car.make)
my_car.fill_tank()
audi
          Fuel tank full!

Modifying Attributes

my_car.fuel_level = 50 # direct
print(my_car.fuel_level)
50

Inheritance

class ElectricCar(Car):
    def __init__(self, make, model, year):
        super().__init__(make, model, year)
        self.battery_size = 40
    def fill_tank(self): # override
        print("No fuel tank!")
my_ecar = ElectricCar('nissan', 'leaf', 2024)
my_ecar.fill_tank()
No fuel tank!

Instances as Attributes

class Battery:
    def __init__(self, size=85):
        self.size = size
        self.charge = 0

class ElectricCar(Car):
    def __init__(...):
        ...
        self.battery = Battery()
my_ecar = ElectricCar('nissan', 'leaf', 2024)
print(my_ecar.battery.size)
85

Storing Objects in a List

gas_fleet = []
for i in range(3):
    gas_fleet.append(Car('ford', 'escape', 2024))
print(f"{len(gas_fleet)} cars ready")
3 cars ready

Understanding self & __init__

  • self — reference to the current object instance
  • __init__() — called automatically when creating an instance
  • Any variable attached to self is available everywhere in the class
  • Class names use CamelCase, objects use lowercase_underscores
