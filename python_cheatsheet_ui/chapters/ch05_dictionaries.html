Dictionaries

Dictionaries store key-value pairs. Use curly braces {}. Each key is connected to a value.

Creating & Accessing

alien = {'color': 'green', 'points': 5}
print(alien['color'])
green

Using get() (safe access)

alien = {'color': 'green'}
print(alien.get('color')) # 'green'
print(alien.get('points', 0)) # 0 (default)
print(alien.get('speed')) # None (no error!)
green
          0
          None

Add / Modify / Delete

alien = {'color': 'green', 'points': 5}
alien['x'] = 0 # add new
alien['color'] = 'yellow' # modify
del alien['points'] # delete
print(alien)
{'color': 'yellow', 'x': 0}

Looping

fav_numbers = {'eric': 7, 'ever': 4, 'erin': 47}

# Key-value pairs
for name, num in fav_numbers.items():
    print(f"{name} loves {num}.")
eric loves 7.
          ever loves 4.
          erin loves 47.
# Keys only
for name in fav_numbers.keys():
    print(name)
eric ever erin
# Values only
for num in fav_numbers.values():
    print(num)
7 4 47

Sorted Keys

for name in sorted(fav_numbers.keys(), reverse=True):
    print(name)
erin ever eric

Length

print(len(fav_numbers))
3

Nesting: List of Dicts

users = []
users.append({'last': 'fermi', 'first': 'enrico'})
users.append({'last': 'curie', 'first': 'marie'})
for u in users:
    print(f"{u['first']} {u['last']}")
enrico fermi
          marie curie

Nesting: Lists in Dict

fav_languages = {
    'jen': ['python', 'ruby'],
    'sarah': ['c'],
}
for name, langs in fav_languages.items():
    print(f"{name}:")
    for lang in langs:
        print(f" - {lang}")
jen:
           - python
           - ruby
          sarah:
           - c

Nesting: Dict of Dicts

users = {
    'aeinstein': {'first': 'albert', 'last': 'einstein'},
    'mcurie': {'first': 'marie', 'last': 'curie'},
}
for uname, info in users.items():
    full = f"{info['first']} {info['last']}"
    print(f"{uname}: {full.title()}")
aeinstein: Albert Einstein
          mcurie: Marie Curie

Dict Comprehensions & zip()

squares = {x:x**2 for x in range(5)}
print(squares)
{0: 0, 1: 1, 2: 4, 3: 9, 4: 16}
group_1 = ['kai', 'abe', 'ada']
group_2 = ['jen', 'eva', 'dan']
pairings = {n:n2 for n, n2 in zip(group_1, group_2)}
print(pairings)
{'kai': 'jen', 'abe': 'eva', 'ada': 'dan'}
