Lists Basics

A list stores items in order. Access items by index, loop through them, add, remove, and modify.

Creating a List

bikes = ['trek', 'redline', 'giant']

Accessing Elements

bikes = ['trek', 'redline', 'giant']
first_bike = bikes[0]
print(first_bike)
trek
last_bike = bikes[-1]
print(last_bike)
giant

Adding Items

bikes = []
bikes.append('trek')
bikes.append('redline')
bikes.append('giant')
print(bikes)
['trek', 'redline', 'giant']

Modifying Items

users = ['val', 'bob', 'mia', 'ron', 'ned']
users[0] = 'valerie'
users[-2] = 'ronald'
print(users)
['valerie', 'bob', 'mia', 'ronald', 'ned']

Removing Items

users = ['val', 'bob', 'mia', 'ron', 'ned']
del users[-1] # delete last
print(users)
['val', 'bob', 'mia', 'ron']
users.remove('mia') # remove by value
print(users)
['val', 'bob', 'ron']
popped = users.pop() # pop last
print(popped)
ron

Sorting

users = ['val', 'bob', 'mia', 'ron', 'ned']
users.sort()
print(users)
['bob', 'mia', 'ned', 'ron', 'val']
users.sort(reverse=True)
print(users)
['val', 'ron', 'ned', 'mia', 'bob']
print(sorted(users)) # temporary sort
['bob', 'mia', 'ned', 'ron', 'val']

Reverse & Length

users = ['val', 'bob', 'mia']
users.reverse()
print(users)
['mia', 'bob', 'val']
print(len(users))
3

Looping

users = ['val', 'bob', 'mia']
for user in users:
    print(f"Welcome, {user}!")
Welcome, val!
          Welcome, bob!
          Welcome, mia!

Inserting

users = ['val', 'bob', 'mia']
users.insert(0, 'joe')
print(users)
['joe', 'val', 'bob', 'mia']
