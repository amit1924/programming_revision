Line / Scatter / Bar

import plotly.express as px
x = list(range(5))
y = [x**2 for x in x]

fig = px.line(x=x, y=y) # line chart
fig.show()
Interactive line chart opens in browser
fig = px.scatter(x=x, y=y) # scatter
fig = px.bar(x=x, y=y) # bar chart
fig.show()
Interactive scatter or bar chart in browser

Titles & Labels

title = "Square Numbers"
labels = {'x': 'Value', 'y': 'Square'}
fig = px.scatter(x=x, y=y, title=title, labels=labels)
fig.show()
Chart with title "Square Numbers" and labeled axes

Advanced Customization

fig = px.scatter(
    x=x, y=y,
    title=title, labels=labels,
    size=y, # bigger dots for bigger values
    color=y, # color by value
    opacity=0.5,
    template='plotly_dark',
)
fig.update_layout(
    title_font_size=30,
    xaxis_title_font_size=24,
)
fig.show()
Dark-themed scatter plot with size/color mapped to values

Subplots & Multiple Traces

from plotly.subplots import make_subplots
import plotly.graph_objects as go

fig = make_subplots(rows=1, cols=2, shared_yaxes=True)
fig.add_trace(go.Scatter(x=x, y=y), row=1, col=1)
cubes = [x**3 for x in x]
fig.add_trace(go.Scatter(x=x, y=cubes), row=1, col=2)
fig.update_layout(title_text="Squares & Cubes")
fig.show()
Two side-by-side charts: Left=squares, Right=cubes

Adding Traces

days = list(range(1, 10))
highs = [60, 63, 68, 70, 68, 70, 66, 62, 64]
lows = [51, 54, 53, 57, 54, 56, 52, 53, 49]
fig = px.line(x=days, y=lows)
fig.add_trace(go.Scatter(x=days, y=highs, mode='lines'))
fig.show()
Two lines: lows (blue) and highs (second trace added)

Geo Maps

lats = [63.069, 60.5671, 46.8529]
lons = [-151.0063, -140.4055, -121.7604]
names = ["Denali", "Mt Logan", "Mt Rainier"]
fig = px.scatter_geo(
    lat=lats, lon=lons,
    title="High Peaks",
    text=names,
    scope="north america",
)
fig.update_traces(textposition="middle right")
fig.show()
Map of North America with labeled mountain peaks
