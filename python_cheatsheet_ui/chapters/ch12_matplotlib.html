Line Graph

import matplotlib.pyplot as plt

x_vals = [0, 1, 2, 3, 4]
squares = [0, 1, 4, 9, 16]
fig, ax = plt.subplots()
ax.plot(x_vals, squares)
plt.show()
Opens a line graph window: x=[0,1,2,3,4], y=[0,1,4,9,16]

Scatter Plot

plt.style.use('seaborn-v0_8')
x_vals = list(range(10))
squares = [x**2 for x in x_vals]
fig, ax = plt.subplots()
ax.scatter(x_vals, squares, s=10)
plt.show()
Scatter plot with seaborn style theme

Customization

ax.set_title('Square Numbers', fontsize=24)
ax.set_xlabel('Value', fontsize=14)
ax.set_ylabel('Square', fontsize=14)
ax.axis([0, 10, 0, 100])
ax.tick_params(axis='both', labelsize=14)
plt.show()
Styled plot with title, axis labels, scaled axes

Colormaps & Emphasis

ax.scatter(x_vals, squares, c=squares, cmap=plt.cm.Blues, s=10)
ax.scatter(x_vals[0], squares[0], c='green', s=100)
ax.scatter(x_vals[-1], squares[-1], c='red', s=100)
Blue colormap, first point green, last point red — larger

Fill Between & Subplots

x_vals = list(range(5))
sq = [x**2 for x in x_vals]
cu = [x**3 for x in x_vals]
ax.fill_between(x_vals, cu, sq, facecolor='blue', alpha=0.25)
plt.show()
Area between squares and cubes filled with blue
# Two subplots sharing x-axis
fig, axs = plt.subplots(2, 1, sharex=True)
axs[0].scatter(x_vals, sq)
axs[1].scatter(x_vals, cu, c='red')
plt.show()
Two plots stacked vertically — squares top, cubes bottom

Working with Dates

from datetime import datetime as dt
new_years = dt.strptime('1/1/2023', '%m/%d/%Y')
today = dt.now()
print(today.strftime('%B %d, %Y'))
May 15, 2026 (current date)
# strptime/strftime codes:
# %A=weekday %B=month %m=month %d=day
# %Y=4-digit year %H=24hr %I=12hr %p=AM/PM
